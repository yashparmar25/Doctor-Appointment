All three folder download nude_module

frondtend and admin folder .env  :

    VITE_BACKEND_URL=
    VITE_RAZORPAY_KEY_ID=
    
backend .env:

    MONGODB_URI=
    CLOUDINARY_NAME =
    CLOUDINARY_API_KEY =
    CLOUDINARY_SECRET_KEY =
    ADMIN_EMAIL =
    ADMIN_PASSWORD =
    JWT_SECRET=
    RAZORPAY_KEY_ID=
    RAZORPAY_KEY_SECRET=

frontend and admin RUN:
      npm run dev

backend RUN:
      npm run server
